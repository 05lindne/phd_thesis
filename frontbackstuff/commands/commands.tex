% create mtd remark environment
\definecolor{graylight}{cmyk}{.30,0,0,.67} % define color using xcolor syntax

  linecolor=graylight,
  topline=false,
  bottomline=false,
  rightline=false,
  skipabove=\topsep,
  skipbelow=\topsep
]{leftrule}
\NewDocumentEnvironment{remark}{O{\textbf{Remark:}}}
{\begin{leftrule}\noindent\textcolor{graylight}{#1}\par}
{\end{leftrule}}
